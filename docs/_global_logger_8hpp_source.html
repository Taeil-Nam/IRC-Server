157 /**
158 * @class LogStream
159 * @brief 로그 메세지를 스트림으로 받는 클래스
160 *
161 * 이 클래스는 RAII 패턴으로 디자인되어 생성과 동시에 소멸을 통해 Log를 생성한다.\n
162 * 연산자 오버로딩(`operator<<`)을 통해 다양한 유형의 데이터를 쉽게 로그 스트림에 추가할 수 있다.\n
163 * `LOG()` 매크로를 사용하면 이 클래스의 인스턴스가 생성되며,\n
164 * 스트림 연산을 통해 로그 메시지를 구성할 수 있다.
165 */
166 class LogStream
167 {
168 public:
169
170 /**
171 * @brief LogStream의 기본 생성자
172 *
173 * 로그 메시지의 상세 정보(로그 레벨, 함수 이름, 파일 이름, 라인 번호)와 함께 로그 스트림을 초기화한다.
174 *
175 * @param level 로그의 심각도
176 * @param functionName 로그가 발생한 함수의 이름
177 * @param fileName 로그가 발생한 파일의 이름
178 * @param lineNumber 로그가 발생한 코드의 라인 번호
179 */
180 LogStream(eSeverityLevel level, const char* functionName,
181 const char* fileName, const int lineNumber);

188 ~LogStream();

190 /**
191 * @brief 스트림에 메세지를 추가하는 연산자 오버로드
192 *
193 * 이 메소드를 통해 다양한 로그 스트림에 대한 데이터 타입의 메세지를 추가할 수 있다.
194 *
195 * @tparam T 데이터 타입
196 * @param message 로그 스트림에 추가할 메세지
197 * @return LogStream& 로그 스트림 객체의 참조를 반환하여 연속 체이닝이 가능하게 한다.
198 */

202 mStream << message;
203 return *this;
